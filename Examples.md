### Simple Video-on-Demand

    rtmp {
        server {
            listen 1935;
            application vod {
                play /var/flvs;
            }
        }
    }

### Simple live broadcast service

    rtmp {
        server {
            listen 1935;
            application live {
                live on;
            }
        }
    }

### Re-translate remote stream

    rtmp {
        server {
            listen 1935;
            application tv {
                live on;
                pull rtmp://cdn.example.com:443/programs/main pageUrl=http://www.example.com/index.html name=maintv;
            }
        }
    }

### Re-translate remote stream to HLS

    rtmp {
        server {
            listen 1935;
            application tv {
                live on;

                hls on;
                hls_path /tmp/tv2;
                hls_fragment 15s;

                pull rtmp://tv2.example.com:443/root/new name=tv2;
            }
        }
    }

    http {
        server {
            listen 80;
            location /tv2 {
                alias /tmp/tv2;
            }
        }
    }
